    for (int i = 0; i < vertex_count; i++)

            closest[i] = start;
        }
    }

    // ÈÄêÊ≠•ÊûÑÂª∫ÊúÄÂ∞èÁîüÊàêÊ†ë
    for (int i = 1; i < vertex_count; i++)

        for (int j = 0; j < vertex_count; j++)

                min_cost = lowcost[j];
                min_index = j;
            }
        }

        // Â∞ÜÊâæÂà∞ÁöÑÈ°∂ÁÇπÂä†ÂÖ•ÊúÄÂ∞èÁîüÊàêÊ†ë
        if (min_index != -1)
